## Kubernetes — Deploy Full Application Stack

apiVersion = app/v1
kind- Deployment
metadata = 
  name: Prabhat-app

  spec:
     replicas = 2       ##tells Deployment how to manage pods
     selector:
       matchLabels:
        app : Prabhat_app
     
     template:       ##how the pods template will look like
       metadata:
         labels:
          app : Prabhat_app

     Spec:
     Containers:            ##tells pods what should be run
        -name : Ngnix
        image : Ngnix
        ports :
          -ContainerPort = 80
